import { useState, useEffect } from 'react';

/**
 * @typedef {Object} Sizes
 * @property {number} width
 * @property {number} height
 */

/**
 * @callback onSizesChange
 * @param {Sizes} sizes 
 */

/**
 * @typedef {Object} ResizeOptions
 * @property {Sizes} sizes
 * @property {onSizesChange} onSizesChange
 * @property {boolean} preserveAspectRatio
 */

/**
 * 
 * @param {ResizeOptions} resizeOptions 
 */
const useResize = ({ sizes, onSizesChange, preserveAspectRatio }) => {

  const [interaction, setInteraction] = useState();

  const handleMouseDown = event => {
    setInteraction({
      startCoordinates: { x: event.pageX, y: event.pageY },
      startSizes: sizes
    });
  };

  useEffect(() => {
    const handleMouseMove = event => {
      if (interaction) {
        const diffX = event.pageX - interaction.startCoordinate.width;
        const diffY= event.pageY - interaction.startCoordinate.height;
        const nextWidth = interaction.startSizes.width + diffX;
        const nextHeight = interaction.startSizes.height + diffY;
        const nextSizes = preserveAspectRatio ?
            Math.max(nextWidth, nextHeight) :
            {
              width: nextWidth,
              height: nextHeight
            };
        onSizesChange(nextSizes);
      }
    };

    const handleMouseUp = () => {
      if (interaction) setInteraction(null);
    };

    document.addEventListener('mousemove', handleMouseMove);
    document.addEventListener('mouseup', handleMouseUp);

    return () => {
      document.removeEventListener('mousemove', handleMouseMove);
      document.removeEventListener('mouseup', handleMouseUp);
    };
  }, [interaction, onSizesChange, preserveAspectRatio]);

  return handleMouseDown;

};

export default useResize;
